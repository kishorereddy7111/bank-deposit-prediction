import streamlit as st
import joblib
import pandas as pd


st.title("Bank Prediction App")
st.write("This app predicts the output using a trained ML model")


model = joblib.load("bank_model.pkl")
scaler = joblib.load("scaler.pkl")
feature_columns = joblib.load("feature_columns.pkl")


st.sidebar.header("Enter Input Values")


input_data = {}

for feature in feature_columns:
    input_data[feature] = st.sidebar.number_input(feature)

input_df = pd.DataFrame([input_data])


input_scaled = scaler.transform(input_df)


prediction = model.predict(input_scaled)


st.subheader("Prediction Result")

if prediction[0] == 1:
    st.success("Positive Outcome")
else:
    st.error("Negative Outcome")
